def initialize(args)
  case args
  when String, Fixnum
    @id = args.to_i
    @href = BASE_URI + id.to_s + '/'
  when Hash
    @href = args[:href]
    @short_name = args[:short_name]
    @name = args[:name]
    @id = args[:id]
  end
end
