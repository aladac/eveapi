def normalize_array_rowset
  self['rowset'].each do |rowset|
    self[rowset['name']] = rowset['row']
  end
  delete('rowset')
end

def normalize_hash_rowset
  self.merge!(self['rowset']['name'] => self['rowset']['row'])
  delete('rowset')
end

def process_rows
  case self['rowset']
  when Hash
    normalize_hash_rowset
  when Array
    normalize_array_rowset
  end
  collapse_key
end
